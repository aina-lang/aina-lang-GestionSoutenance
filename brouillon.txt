// Récupération du bouton et du modal

var modal = document.getElementById("mon-modal");

// Récupération du span pour fermer le modal
var span = document.getElementsByClassName("close")[0];

// Ajout d'un événement au clic sur le bouton
bouton.addEventListener("click", function() {
  // Affichage du modal
  modal.style.display = "block";

  // Récupération du matricule de l'étudiant
  var matricule = {{ $etudiant->matricule}}; // Remplacer par le matricule de l'étudiant

  // Affichage du matricule dans le modal
  var matriculeElement = document.getElementById("matricule-etudiant");
  matriculeElement.innerHTML = "Matricule de l'étudiant : " + matricule;
});

// Ajout d'un événement au clic sur le span pour fermer le modal
span.addEventListener("click", function() {
  modal.style.display = "none";
});
</script>
